```typescript
interface TranscriptionConfig {
  model: 'whisper-1';
  language?: string;  // Auto-detect if not specified
  prompt?: string;    // Context for better accuracy

  // Sport-specific vocabulary
  vocabulary: {
    tennis: ['forehand', 'backhand', 'volley', 'serve', 'toss', 'rally', 'baseline'],
    swimming: ['freestyle', 'backstroke', 'breaststroke', 'butterfly', 'flip turn'],
    // ... more sports
  };
}

async function transcribeVoiceNote(
  audioUrl: string,
  sportType: string
): Promise<Transcription> {
  const vocabulary = TRANSCRIPTION_CONFIG.vocabulary[sportType] || [];

  const response = await openai.audio.transcriptions.create({
    file: await fetchAudioFile(audioUrl),
    model: 'whisper-1',
    prompt: `Sports coaching session feedback about ${sportType}. Technical terms: ${vocabulary.join(', ')}`,
    response_format: 'verbose_json',
    timestamp_granularities: ['word'],
  });

  return {
    text: response.text,
    words: response.words,
    duration: response.duration,
    language: response.language,
  };
}
```

```typescript
const ENHANCEMENT_PROMPT = `
You are helping a sports coach write better session feedback for parents.

CONTEXT:
- Sport: {sport_type}
- Student: {student_name}, age {student_age}
- Session type: {session_type} ({duration} minutes)
- Previous session notes: {previous_notes}
- Student's current skill levels: {skill_levels}

COACH'S RAW INPUT:
{raw_notes}

RATINGS PROVIDED:
- Overall: {overall_rating}/5
- Effort: {effort_rating}/5
- Focus: {focus_rating}/5

YOUR TASK:
1. Transform the raw notes into a professional, structured summary
2. Maintain the coach's voice and personality
3. Be specific about what was worked on and progress made
4. Keep it encouraging but honest
5. Structure with clear sections
6. Keep it concise (150-250 words)

RULES:
- NEVER add information not in the original notes
- NEVER make up specific statistics or numbers
- NEVER provide medical or injury advice
- NEVER compare to other students
- Use the student's name naturally

OUTPUT JSON:
{
  "enhanced_summary": "The main feedback text",
  "key_insights": ["Point 1", "Point 2", "Point 3"],
  "areas_for_improvement": ["Area 1"],
  "parent_summary": "2-3 sentence summary for parents"
}
`;

async function enhanceFeedback(input: FeedbackInput): Promise<EnhancedFeedback> {
  const prompt = ENHANCEMENT_PROMPT
    .replace('{sport_type}', input.sportType)
    .replace('{student_name}', input.studentName)
    // ... replace all placeholders

  const response = await openai.chat.completions.create({
    model: 'gpt-4-turbo-preview',
    messages: [
      { role: 'system', content: prompt },
      { role: 'user', content: input.rawNotes }
    ],
    response_format: { type: 'json_object' },
    temperature: 0.7,
    max_tokens: 1000,
  });

  return JSON.parse(response.choices[0].message.content);
}
```

```typescript
  // Get drill library embeddings
  const drillEmbeddings = await getDrillEmbeddings(sportType);

  // Create query from skill gaps
  const queryText = skillGaps
    .map(g => `${g.skillName}: ${g.notes}`)
    .join('. ');

  const queryEmbedding = await createEmbedding(queryText);

  // Find similar drills
  const similarDrills = await searchSimilarDrills(
    queryEmbedding,
    drillEmbeddings,
    { limit: 5, minScore: 0.7 }
  );

  // Filter by difficulty level
  const appropriateDrills = similarDrills.filter(drill =>
    drill.difficultyLevel <= getPlayerLevel(playerId, sportType) + 1
  );

  return appropriateDrills.slice(0, 3);
}
```

```typescript
  const asset = await mux.video.assets.create({
    input: uploadUrl,
```
